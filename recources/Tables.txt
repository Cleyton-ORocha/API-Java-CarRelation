CREATE TABLE Owner(
    IDOwner INT PRIMARY KEY AUTO_INCREMENT,
    Name VARCHAR(100) NOT NULL,
    CPF CHAR(11) NOT NULL UNIQUE,
    Age INT(3) NOT NULL,
    Gender ENUM("Masculine", "Feminine") NOT NULL
);

CREATE TABLE Car(
    IDCar INT PRIMARY KEY AUTO_INCREMENT,
    Year INT(4) NOT NULL,
    Price FLOAT(10,2) NOT NULL,
    Color VARCHAR(12) NOT NULL,
    Brand VARCHAR(20) NOT NULL,
    FactoryName VARCHAR(50) NOT NULL, 
    ID_Owner INT
);

CREATE TABLE Address(
    IDAddress INT PRIMARY KEY AUTO_INCREMENT,
    State CHAR(2) NOT NULL,
    City VARCHAR(50) NOT NULL, 
    District VARCHAR(50) NOT NULL, 
    Street VARCHAR(100) NOT NULL, 
    ID_Owner INT
);

CREATE TABLE Phone(
    IDPhone INT PRIMARY KEY AUTO_INCREMENT,
    Numero VARCHAR(25) NOT NULL UNIQUE,
    ID_Owner INT
);

ALTER TABLE Car
ADD CONSTRAINT FK_Owner_Car
FOREIGN KEY (ID_Owner)
REFERENCES Owner(IDOwner);

ALTER TABLE Address
ADD CONSTRAINT FK_Owner_Address
FOREIGN KEY (ID_Owner)
REFERENCES Owner(IDOwner);

ALTER TABLE Phone
ADD CONSTRAINT FK_Owner_Phone
FOREIGN KEY (ID_Owner)
REFERENCES Owner(IDOwner);
